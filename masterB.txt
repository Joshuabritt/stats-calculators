function renderGrid(){
  let h='<tr><th>#</th>' + state.headers.map((v,i)=>`<th data-col="${i}" class="${state.selectedCols.includes(i)?'selected':''}">${v}</th>`).join('') + '</tr>';

    h += `<tr><th>${r+1}</th>` + Array.from({length:state.cols},(_,c)=>`<td contenteditable="true" data-r="${r}" data-c="${c}"></td>`).join('') + '</tr>';
  }
  grid.innerHTML = h;
  grid.querySelectorAll('th[data-col]').forEach(th=>th.onclick=e=>{
    const col=+th.dataset.col;
    if(e.ctrlKey||e.metaKey){
      state.selectedCols = state.selectedCols.includes(col) ? state.selectedCols.filter(c=>c!==col) : [...state.selectedCols,col].sort((a,b)=>a-b);

  const w=720,h=280,p=30,left=mu-4*sd,right=mu+4*sd,X=v=>p+(v-left)/(right-left)*(w-2*p),pdf=v=>Math.exp(-0.5*((v-mu)/sd)**2)/(sd*Math.sqrt(2*Math.PI)),max=pdf(mu),Y=v=>h-p-(v/max)*(h-2*p);

function computeNormalInline(){
  const f=normalForm;
  const modeEl=f.querySelector('input[name="mode"]:checked');
  const mode=modeEl?modeEl.value:'standard';
  const mu=+f.mu.value, sd=+f.sd.value;
  if(!(sd>0)) throw 'Standard deviation must be positive.';
  const op=f.op.value;
  const normProb=v=>{let p=+v; if(p>1)p/=100; return p;};
  const z=v=>(v-mu)/sd;
  if(mode==='standard'){
    const xTxt=(f.x.value||'').trim(), pTxt=(f.p.value||'').trim();
    if(xTxt && !pTxt){
      const x=+xTxt; if(!Number.isFinite(x)) throw 'Enter a valid x value.';
      const p=op==='<='?normCdf(z(x)):(1-normCdf(z(x)));
      f.p.value=fmt(p,8);
      updateNormalGraphInline(mu,sd,op==='<='?mu-4*sd:x,op==='<='?x:mu+4*sd);
      return;
    }
    if(pTxt && !xTxt){
      const p=normProb(pTxt); if(!(p>0&&p<1)) throw 'Probability must be in (0,1) or (0,100).';
      const x=op==='<='?(mu+sd*normInv(p)):(mu+sd*normInv(1-p));
      f.x.value=fmt(x,8);
      updateNormalGraphInline(mu,sd,op==='<='?mu-4*sd:x,op==='<='?x:mu+4*sd);
      return;
    }
    if(xTxt&&pTxt){
      const x=+xTxt, p=op==='<='?normCdf(z(x)):(1-normCdf(z(x)));
      f.p.value=fmt(p,8);
      updateNormalGraphInline(mu,sd,op==='<='?mu-4*sd:x,op==='<='?x:mu+4*sd);
      return;
    }
    throw 'Enter x or probability.';
  }
  const loTxt=(f.lo.value||'').trim(), hiTxt=(f.hi.value||'').trim(), pMidTxt=(f.pmid.value||'').trim();
  if(loTxt && hiTxt && !pMidTxt){
    const lo=+loTxt, hi=+hiTxt; if(!(hi>lo)) throw 'Upper bound must be greater than lower bound.';
    const p=normCdf(z(hi))-normCdf(z(lo));
    f.pmid.value=fmt(p,8);
    updateNormalGraphInline(mu,sd,lo,hi);
    return;
  }
  if(pMidTxt && !loTxt && !hiTxt){
    const pmid=normProb(pMidTxt); if(!(pmid>0&&pmid<1)) throw 'Between probability must be in (0,1) or (0,100).';
    const zStar=normInv((1+pmid)/2), lo=mu-sd*zStar, hi=mu+sd*zStar;
    f.lo.value=fmt(lo,8); f.hi.value=fmt(hi,8);
    updateNormalGraphInline(mu,sd,lo,hi);
    return;
  }
  if(loTxt&&hiTxt&&pMidTxt){
    const lo=+loTxt, hi=+hiTxt; if(!(hi>lo)) throw 'Upper bound must be greater than lower bound.';
    updateNormalGraphInline(mu,sd,lo,hi);
    return;
  }
  throw 'Enter lower & upper OR enter between probability only.';
}
